        async function destekListesiGuncelle() {
            let fd = new FormData(); fd.append('islem', 'get_conversations');
            let res = await fetch('/api/destek-api.php', { method: 'POST', body: fd });
            let data = await res.json();
            
            let html = "";
            data.forEach(d => {
                let activeClass = (aktifDestekUser === d.gonderen) ? 'active' : '';
                html += `<div class="user-list-item ${activeClass}" onclick="kullaniciSohbetAc('${d.gonderen}')">👤 ${d.gonderen}</div>`;
            });
            document.getElementById('destekUserList').innerHTML = html || "<div style='padding:20px; text-align:center;'>Kimse mesaj atmadı.</div>";
        }

        function kullaniciSohbetAc(user) { 
            aktifDestekUser = user; 
            document.getElementById('activeDestekUser').innerText = "Kullanıcı: " + user; 
            destekMesajlariGetir(user); 
        }

        async function destekMesajlariGetir(user) {
            let fd = new FormData(); fd.append('islem', 'get_messages'); fd.append('user1', user); fd.append('user2', 'admin');
            let res = await fetch('/api/destek-api.php', { method: 'POST', body: fd });
            let msgs = await res.json();
            
            let html = "";
            msgs.forEach(m => {
                let isMe = m.gonderen === 'admin';
                let styleClass = isMe ? 'admin' : 'user'; 
                let dosyaHtml = "";
                if(m.dosya) dosyaHtml = `<br><a href="/${m.dosya}" target="_blank" class="s-file-link">📎 Ekli Dosya</a>`;
                
                html += `<div class="s-msg ${styleClass}"><b>${isMe ? 'Sen' : m.gonderen}</b><br>${m.mesaj} ${dosyaHtml} <div style="font-size:10px; opacity:0.7; margin-top:5px; text-align:right;">${m.tarih}</div></div>`;
            });
            
            let b = document.getElementById('destekChatBody');
            let isAtBottom = b.scrollHeight - b.clientHeight <= b.scrollTop + 15;
            b.innerHTML = html;
            if(isAtBottom) b.scrollTop = b.scrollHeight;
        }

        async function adminDestekGonder() {
            if(!aktifDestekUser) return alert("Önce soldan bir kullanıcı seçin.");
            const input = document.getElementById('destekAdminInput');
            const fileInput = document.getElementById('destekAdminFile');
            const mesaj = input.value.trim();
            const dosya = fileInput.files[0];

            if(!mesaj && !dosya) return;

            let fd = new FormData();
            fd.append('islem', 'send'); fd.append('gonderen', 'admin'); fd.append('alici', aktifDestekUser); fd.append('mesaj', mesaj);
            if(dosya) fd.append('dosya', dosya);

            input.value = ''; fileInput.value = '';
            await fetch('/api/destek-api.php', { method: 'POST', body: fd });
            destekMesajlariGetir(aktifDestekUser);
        }

        let tumIslemKayitlari = []; 
        async function kayitlariGetir() { 
            let res = await fetch('/api/admin-api.php'); 
            let data = await res.json(); 
            tumIslemKayitlari = data; 
            let tbody = document.getElementById('adminTableBody'); 
            if(!data || data.length === 0) return tbody.innerHTML = '<tr><td colspan="4" style="text-align: center;">Kayıt bulunamadı.</td></tr>'; 
            
            let html = ""; 
            data.forEach(k => { 
                html += `<tr>
                    <td>${k.islemTarihi}</td>
                    <td><span class="badge">${k.islemYapan}</span></td>
                    <td><strong>${k.mizanDonemi}</strong></td>
                    <td style="text-align: right;"><button class="btn-action btn-view" onclick="mizanDetayGoster(${k.id})">👁️ Gör</button></td>
                </tr>`; 
            }); 
            tbody.innerHTML = html; 
        }

        async function kayitlariTemizle() { 
            if(!confirm("Tüm mizan kayıtlarını silmek istediğinize emin misiniz?")) return;
            await fetch('/api/admin-api.php', { method: 'POST', body: JSON.stringify({ islem: "temizle" }) }); 
            kayitlariGetir(); 
        }

        function mizanDetayGoster(kayitId) {
            const kayit = tumIslemKayitlari.find(k => Number(k.id) === Number(kayitId));
            if(!kayit) return alert("Kayıt bulunamadı.");
            let mizanVerisi = kayit.orijinalMizan;
            if (typeof mizanVerisi === 'string') { try { mizanVerisi = JSON.parse(mizanVerisi); } catch(e) { mizanVerisi = []; } }
            
            document.getElementById('detayAltBaslik').innerText = `${kayit.islemYapan} | Dönem: ${kayit.mizanDonemi}`;
            let html = "";
            mizanVerisi.forEach(m => {
                if(m.no !== undefined) {
                    html += `<tr><td style="font-family:monospace; font-weight:bold;">${m.no}</td><td>${m.ad}</td><td style="text-align:right;">${m.borc}</td><td style="text-align:right;">${m.alacak}</td><td style="text-align:right;">${m.borcBakiye}</td><td style="text-align:right;">${m.alacakBakiye}</td></tr>`;
                }
            });
            document.getElementById('mizanTableBody').innerHTML = html || `<tr><td colspan="6" style="text-align:center; color:#ef4444;">Veri uyumsuzluğu.</td></tr>`;
            document.getElementById('mizanModal').style.display = 'flex';
        }

        function excelOlarakIndir() {
            const table = document.getElementById("exportTable"); let html = table.outerHTML;
            const uri = 'data:application/vnd.ms-excel;base64,';
            const template = '<html xmlns:o="urn:schemas-microsoft-com:office:office" xmlns:x="urn:schemas-microsoft-com:office:excel" xmlns="http://www.w3.org/TR/REC-html40"><head><meta charset="UTF-8"></head><body>{table}</body></html>';
            const base64 = function(s) { return window.btoa(unescape(encodeURIComponent(s))) };
            const format = function(s, c) { return s.replace(/{(\w+)}/g, function(m, p) { return c[p]; }) };
            const ctx = { worksheet: 'OrijinalMizan', table: html };
            const a = document.createElement('a'); a.href = uri + base64(format(template, ctx)); a.download = 'Mizan_Tablosu.xls'; document.body.appendChild(a); a.click(); document.body.removeChild(a);
        }

        async function kullanicilariYukle() { 
            let res = await fetch('/api/auth-api.php'); let data = await res.json(); let tbody = document.getElementById('kullaniciTableBody'); 
            let html = ""; 
            data.forEach(u => { 
                if (u.username === 'admin') return; 
                const isBanned = (u.is_banned == 1); 
                const kycStatus = u.kyc_status;
                
                let durumGorseli = '<span class="badge" style="background:#dcfce7; color:#15803d;">🟢 Aktif</span>';
                if(isBanned) durumGorseli = '<span class="badge-danger">⛔ Yasaklı</span>';
                else if(kycStatus == 1) durumGorseli = '<span class="badge" style="background:#fef08a; color:#a16207;">⚠️ Uyarıldı</span>';
                else if(kycStatus == 2) durumGorseli = '<span class="badge" style="background:#bae6fd; color:#0369a1;">⏳ Bekliyor</span>';

                html += `<tr>
                    <td><strong>${u.username}</strong><br><small style="color:#64748b;">${u.kooperatif || 'Bilinmiyor'}</small></td>
                    <td>${durumGorseli}</td>
                    <td style="text-align: right; display:flex; gap:5px; justify-content:flex-end;">
                        <button class="btn-action btn-warn" title="Uyarı At" onclick="uyariGonder('${u.username}')">⚠️ Uyarı</button>
                        ${isBanned ? `<button class="btn-action btn-unban" onclick="kullaniciBanla('${u.username}', false)">✔️ Aç</button>` : `<button class="btn-action btn-ban" onclick="kullaniciBanla('${u.username}', true)">⛔ Banla</button>`}
                        <button class="btn-action btn-edit" title="Şifre Değiştir" onclick="duzenleModaliniAc('${u.username}')">🔑</button>
                        <button class="btn-action btn-del" title="Tamamen Sil" onclick="kullaniciSil('${u.username}')">🗑️</button>
                    </td>
                </tr>`; 
            }); 
            tbody.innerHTML = html || '<tr><td colspan="3" style="text-align:center;">Kullanıcı Yok</td></tr>'; 
        }

        async function uyariGonder(user) {
            if(!confirm(`'${user}' isimli kullanıcının ekranını kilitleyip gerçek bilgilerini girmesini istemekte emin misiniz?`)) return;
            await fetch('/api/auth-api.php', { method: 'POST', body: JSON.stringify({ islem: "kyc_uyari_gonder", username: user }) });
            kullanicilariYukle(); alert("Uyarı gönderildi, kullanıcının ekranı kilitlendi.");
        }

        async function kullaniciBanla(username, statu) { await fetch('/api/auth-api.php', { method: 'POST', body: JSON.stringify({ islem: "hesap_banla", username: username, statu: statu }) }); kullanicilariYukle(); }
        async function kullaniciSil(username) { if(confirm("Emin misiniz?")) { await fetch('/api/auth-api.php', { method: 'POST', body: JSON.stringify({ islem: "hesap_sil", username: username }) }); kullanicilariYukle(); } }

        async function kullaniciyiKaydet() {
            const eski = document.getElementById('eskiKullaniciAdi').value; const yeni = document.getElementById('yeniKullaniciAdi').value.trim(); const sifre = document.getElementById('yeniSifre').value.trim();
            if (!yeni) return alert("Kullanıcı adı boş olamaz!");
            await fetch('/api/auth-api.php', { method: 'POST', body: JSON.stringify({ islem: "kullanici_guncelle", eskiKullaniciAdi: eski, yeniKullaniciAdi: yeni, yeniSifre: sifre }) });
            document.getElementById('editModal').style.display='none'; kullanicilariYukle(); alert("Bilgiler güncellendi!");
        }

        async function hesapEkleTekli() {
            const ad = document.getElementById('yeniHesapAd').value.trim(); const sifre = document.getElementById('yeniHesapSifre').value.trim();
            if(!ad || !sifre) return alert("Boş alan bırakmayın!");
            await fetch('/api/auth-api.php', { method: 'POST', body: JSON.stringify({ islem: "admin_hesap_ekle", hesaplar: [{kullanici: ad, sifre: sifre}] }) });
            document.getElementById('hesapEkleModal').style.display='none'; document.getElementById('yeniHesapAd').value = ''; document.getElementById('yeniHesapSifre').value = '';
            kullanicilariYukle();
        }

        async function kycİstekleriniGetir() {
            let res = await fetch('/api/auth-api.php', { method: 'POST', body: JSON.stringify({ islem: "get_pending_kyc" }) });
            let data = await res.json();
            const btn = document.getElementById('kycBadgeBtn');
            if(data.length > 0) { btn.style.display = 'block'; btn.className = 'btn-action btn-del pulse-btn'; btn.innerText = `📋 İstekler (${data.length})`; } 
            else { btn.style.display = 'none'; document.getElementById('kycModal').style.display = 'none'; }
            let html = "";
            data.forEach(d => {
                html += `
                <div class="req-box">
                    <div class="req-info">
                        <span style="color:#ef4444; font-weight:bold; font-size:11px;">ESKİ BİLGİ:</span><br>
                        ${d.username} <br> <small>${d.kooperatif || 'Yok'}</small><br><br>
                        <span style="color:#10b981; font-weight:bold; font-size:11px;">YENİ İSTENEN BİLGİ:</span><br>
                        <strong>${d.pending_name}</strong> <br> <small>${d.pending_koop}</small>
                    </div>
                    <div style="display:flex; flex-direction:column; gap:5px;">
                        <button class="btn-action btn-unban" onclick="kycIslem('${d.username}', 'onayla')">✔️ Onayla</button>
                        <button class="btn-action btn-del" onclick="kycIslem('${d.username}', 'reddet')">✖ Reddet</button>
                    </div>
                </div>`;
            });
            document.getElementById('kycListBody').innerHTML = html || "<div style='text-align:center; padding:20px; color:#64748b;'>Bekleyen istek yok.</div>";
        }

        async function kycIslem(user, tip) {
            let islemTipi = (tip === 'onayla') ? 'kyc_onayla' : 'kyc_reddet';
            await fetch('/api/auth-api.php', { method: 'POST', body: JSON.stringify({ islem: islemTipi, username: user }) });
            kycİstekleriniGetir(); kullanicilariYukle();
        }
