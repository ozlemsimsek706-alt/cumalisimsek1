        function switchTab(tab) {
            // TARAYICI CİHAZ KONTROLÜ
            if(tab === 'register' && localStorage.getItem('tk_cihaz_kayitli')) {
                mesajGoster("Güvenlik Kilidi: Bu bilgisayardan zaten bir hesap oluşturdunuz. Yeni bir kayıt yapamazsınız.", "error");
                return;
            }

            document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
            document.querySelectorAll('.form-view').forEach(v => v.classList.remove('active'));
            document.getElementById('mesajKutusu').style.display = 'none';

            if(tab === 'login') {
                document.querySelectorAll('.tab-btn')[0].classList.add('active');
                document.getElementById('loginView').classList.add('active');
            } else {
                document.querySelectorAll('.tab-btn')[1].classList.add('active');
                document.getElementById('registerView').classList.add('active');
            }
        }

        function mesajGoster(mesaj, tur) {
            const kutu = document.getElementById('mesajKutusu');
            kutu.className = `alert-box alert-${tur}`;
            kutu.innerText = mesaj;
            kutu.style.display = 'block';
        }

        async function girisYap() {
            const k = document.getElementById('lKullanici').value.trim();
            const s = document.getElementById('lSifre').value.trim();

            if(!k || !s) return mesajGoster("Lütfen kullanıcı adı ve şifreyi doldurun.", "error");

            try {
                const res = await fetch('/api/auth-api.php', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ islem: "login", kullanici: k, sifre: s })
                });
                const data = await res.json();
                if(res.ok && data.success) {
                    localStorage.setItem('tk_oturum', k);
                    localStorage.setItem('tk_is_admin', data.isAdmin);
                    window.location.href = 'index.html';
                } else {
                    mesajGoster(data.error || "Giriş başarısız.", "error");
                }
            } catch (e) { mesajGoster("Sunucuya bağlanılamadı.", "error"); }
        }

        async function kayitOl() {
            const ad = document.getElementById('rAdSoyad').value.trim();
            const koop = document.getElementById('rKoop').value;
            const sifre = document.getElementById('rSifre').value.trim();

            if(!ad || !koop || !sifre) return mesajGoster("Lütfen Ad Soyad, Kooperatif ve Şifre alanlarının tümünü doldurun.", "error");

            try {
                const res = await fetch('/api/auth-api.php', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ islem: "register", adSoyad: ad, kooperatif: koop, sifre: sifre })
                });
                const data = await res.json();
                if(res.ok && data.success) {
                    // BAŞARILI KAYIT SONRASI CİHAZI KALICI OLARAK İŞARETLE
                    localStorage.setItem('tk_cihaz_kayitli', 'true');
                    
                    mesajGoster("Kayıt başarılı! Şimdi belirlediğiniz Ad Soyad ve şifre ile giriş yapabilirsiniz.", "info");
                    document.getElementById('lKullanici').value = ad;
                    setTimeout(() => switchTab('login'), 2000);
                } else {
                    mesajGoster(data.error || "Kayıt olurken bir hata oluştu.", "error");
                }
            } catch (e) { mesajGoster("Sunucuya bağlanılamadı.", "error"); }
        }
